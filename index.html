        function addSingleStudent() {

            const nameInput = document.getElementById('single-student-name');

            const name = nameInput.value.trim();


            if (!name) {

                showToast("Nama tidak boleh kosong!", "alert-triangle", "rose");

                return;

            }


            const isDuplicate = students.some(s => s.name.toLowerCase() === name.toLowerCase());

            if (isDuplicate) {

                showToast("Nama siswa tersebut sudah ada di daftar!", "alert-triangle", "rose");

                return;

            }


            const newStudent = {

                id: crypto.randomUUID(),

                name: name

            };


            students.push(newStudent);

            saveToLocalStorage();

            nameInput.value = "";

            renderApp();

            showToast(`${name} berhasil didaftarkan!`, "user-check", "emerald");

        }

        function calculateReports() {

            const reportTbody = document.getElementById('report-tbody');

            if (students.length === 0) {

                reportTbody.innerHTML = `

                    <tr>

                        <td colspan="7" class="px-6 py-8 text-center text-slate-400">

                            <div class="flex flex-col items-center space-y-2">

                                <p>Belum ada data untuk kalkulasi laporan.</p>

                            </div>

                        </td>

                    </tr>

                `;

                return;

            }


            // Count total tracked days (dates that have at least 1 attendance record)

            const dates = Object.keys(attendanceDb).filter(date => {

                return Object.keys(attendanceDb[date]).length > 0;

            });


            document.getElementById('total-days-tracked').innerText = `${dates.length} Hari`;


            let bestStudent = "-";

            let maxPresentRatio = -1;


            let worstStudent = "-";

            let worstAttendancePoints = -1; // Heavy points for absent/sakit


            const rowsHtml = students.map((student, idx) => {

                let sHadir = 0;

                let sSakit = 0;

                let sIzin = 0;

                let sAlpa = 0;


                dates.forEach(date => {

                    const status = attendanceDb[date][student.id];

                    if (status === 'Hadir') sHadir++;

                    if (status === 'Sakit') sSakit++;

                    if (status === 'Izin') sIzin++;

                    if (status === 'Alpa') sAlpa++;

                });


                const totalAttendedOrNoEx = sHadir + sSakit + sIzin + sAlpa;

                let pct = 0;

                if (totalAttendedOrNoEx > 0) {

                    pct = Math.round((sHadir / totalAttendedOrNoEx) * 100);

                } else if (dates.length > 0) {

                    pct = 0; // Days tracked but this student was untracked

                } else {

                    pct = 100; // No tracking yet, clean slate

                }


                // Analytics for high performer (Most presents)

                if (sHadir > maxPresentRatio || (sHadir === maxPresentRatio && maxPresentRatio > 0)) {

                    maxPresentRatio = sHadir;

                    bestStudent = student.name;

                }


                // Analytics for attention seeker (Most Alpa/Sakit/Izin sum)

                const pointLoss = (sAlpa * 3) + (sSakit * 1) + (sIzin * 1);

                if (pointLoss > worstAttendancePoints && pointLoss > 0) {

                    worstAttendancePoints = pointLoss;

                    worstStudent = student.name;

                }


                // Percentage Badge style logic

                let pctColor = "bg-rose-50 text-rose-700 border-rose-100";

                if (pct >= 90) pctColor = "bg-emerald-50 text-emerald-700 border-emerald-100";

                else if (pct >= 75) pctColor = "bg-indigo-50 text-indigo-700 border-indigo-100";

                else if (pct >= 50) pctColor = "bg-amber-50 text-amber-700 border-amber-100";


                return `

                    <tr class="hover:bg-slate-50/50 transition">

                        <td class="px-6 py-4 text-center font-bold text-slate-400">${idx + 1}</td>

                        <td class="px-6 py-4 font-semibold text-slate-900">${student.name}</td>

                        <td class="px-6 py-4 text-center font-medium text-emerald-600">${sHadir}</td>

                        <td class="px-6 py-4 text-center font-medium text-blue-600">${sSakit}</td>

                        <td class="px-6 py-4 text-center font-medium text-amber-500">${sIzin}</td>

                        <td class="px-6 py-4 text-center font-medium text-rose-600">${sAlpa}</td>

                        <td class="px-6 py-4 text-center">

                            <span class="px-2.5 py-1 text-xs font-bold rounded-full border ${pctColor}">${pct}%</span>

                        </td>

                    </tr>

                `;

            }).join('');


            reportTbody.innerHTML = rowsHtml;

            document.getElementById('top-student').innerText = bestStudent !== "-" ? bestStudent : "Belum terdata";

            document.getElementById('attention-student').innerText = worstStudent !== "-" ? worstStudent : "Semua hadir aman";


            lucide.createIcons();

        }
